create table cash_operation (user_email text, ammount real, type text, auth_num text, date timestamp, bank text, is_procesed boolean );
create table registered_user (user_email text, full_name text, status text, registered_date timestamp);

create table event_browser (auth_num text, history_elements json, user_page_selected json);

CREATE OR REPLACE FUNCTION update_cash_operation() RETURNS TRIGGER AS
$BODY$
BEGIN
    UPDATE cash_operation SET is_procesed = true WHERE auth_num = new.auth_num;
    RETURN new;
END;
$BODY$
language plpgsql;

CREATE TRIGGER new_event_browser
    AFTER INSERT ON event_browser
    FOR EACH ROW
    EXECUTE PROCEDURE update_cash_operation();
